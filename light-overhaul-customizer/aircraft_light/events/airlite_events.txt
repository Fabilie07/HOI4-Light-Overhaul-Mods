add_namespace = airlite_tech_pack

country_event = { # Any 1933 Airframe
	id = airlite_tech_pack.1
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				has_game_rule = { airlite_tech_unlock_rule = TECHNOLOGY }
				
				OR = {
					has_tech = iw_small_airframe
					has_tech = iw_medium_airframe
					has_tech = iw_large_airframe
					has_tech = bba_early_transport_plane
				}
			}
			
			AND = {
				has_game_rule = { airlite_tech_unlock_rule = TIME }
				
				date > 1933.01.01
			}
		}
		
		NOT = { has_tech = airlite_wings_1 }
	}

	immediate = {
		set_technology = { airlite_wings_1 = 1 popup = no }
		set_technology = { airlite_wheels_1 = 1 popup = no }
		set_technology = { airlite_frame_1 = 1 popup = no }
	}
}
